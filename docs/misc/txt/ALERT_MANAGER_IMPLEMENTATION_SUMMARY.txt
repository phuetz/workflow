QUICK START
===========

1. Import:
   import { alertManager } from './monitoring/AlertManager';

2. Initialize:
   import { setupAlertManager } from './monitoring/alertManagerConfig.example';
   await setupAlertManager();

3. Create Alert:
   const alert = await alertManager.createAlert({
     title: 'Critical Event',
     severity: 'critical',
     source: 'system',
     category: 'security',
     description: 'Description',
     recommended_actions: ['Action 1', 'Action 2']
   });

4. Manage:
   await alertManager.acknowledgeAlert(alert.id, 'user@company.com');
   await alertManager.resolveAlert(alert.id, 'user@company.com');

5. Analyze:
   const stats = alertManager.getAlertStats({
     start: new Date(Date.now() - 24*60*60*1000),
     end: new Date()
   });
